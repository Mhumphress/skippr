// ============================================================
//  SKIPPR CONSULT INTAKE — Google Apps Script (Standalone)
//
//  This is a STANDALONE Apps Script project, separate from the
//  container-bound Skippr CRM script. It targets the same CRM
//  spreadsheet via openById, writes new leads directly into the
//  Pipeline sheet (so they appear in the CRM PWA at crm.skippr.us),
//  and exposes a doPost(e) Web App endpoint that both the consult
//  form and the homepage contact form post to.
//
//  Deploy: Apps Script editor → Deploy → New deployment →
//          Web app → Execute as: me, Who has access: Anyone.
// ============================================================

// ── ENTRY POINT ─────────────────────────────────────────────
function doPost(e) {
  try {
    const params = (e && e.parameter) || {};
    const result = handleConsultSubmission(params);
    return ContentService
      .createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    Logger.log("doPost error: " + err + "\n" + (err.stack || ""));
    return ContentService
      .createTextOutput(JSON.stringify({ ok: false, error: String(err) }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// ── ORCHESTRATOR ────────────────────────────────────────────
function handleConsultSubmission(params) {
  // Honeypot: if bots filled the hidden field, silently discard.
  if (params.company_website && String(params.company_website).trim() !== "") {
    Logger.log("Honeypot triggered; submission silently dropped.");
    return { ok: true };
  }

  // Origin check.
  const origin = String(params._origin || "").trim();
  if (CONFIG.ALLOWED_ORIGINS.indexOf(origin) === -1) {
    return { ok: false, error: "Bad origin: " + origin };
  }

  const data = normalizeFormData(params);

  const validation = validateLeadData(data);
  if (!validation.ok) return validation;

  data.tier      = parseTier(data.serviceOrPackage);
  data.firstName = firstNameOf(data.name);
  data.timestamp = new Date();

  const ss = SpreadsheetApp.openById(CONFIG.SHEET_ID);
  data.leadId = appendToPipeline(ss, data);

  sendAutoReply(data);
  sendInternalNotification(data);

  return { ok: true, id: data.leadId };
}

// ── SETUP CHECK ─────────────────────────────────────────────
// Run this once after pasting the script. Verifies the target CRM
// spreadsheet has the Pipeline + Settings tabs (which the main CRM
// script's setupCRM() creates). Does NOT create any new tabs here.
function setup() {
  const ss = SpreadsheetApp.openById(CONFIG.SHEET_ID);
  const missing = [];
  if (!ss.getSheetByName("Pipeline")) missing.push("Pipeline");
  if (!ss.getSheetByName("Settings")) missing.push("Settings");
  if (missing.length) {
    throw new Error("Missing sheet(s) in CRM spreadsheet: " + missing.join(", ") +
                    ". Run the main CRM's setupCRM() first.");
  }
  Logger.log("✅ Setup check passed. CRM spreadsheet has Pipeline and Settings tabs.");
  Logger.log("Note: the old 📥 Leads tab is no longer used by this script — safe to delete from the spreadsheet.");
}

// ── PIPELINE APPEND ─────────────────────────────────────────
// Appends a new lead row to the Pipeline sheet in the same
// 37-column schema the main CRM uses (see buildPipeline in CRM.gs).
// Generates a fresh SKC ID under a DocumentLock so concurrent
// submissions + manual CRM use can't collide on the counter.
function appendToPipeline(ss, data) {
  const pipeline = ss.getSheetByName("Pipeline");
  if (!pipeline) throw new Error("Pipeline sheet not found. Run setup() first.");

  const id = reserveNextSkcId(ss, data.tier);

  const row = new Array(37).fill("");
  row[0]  = id;                         // A  — Customer ID
  row[1]  = data.name;                  // B  — Customer Name
  row[3]  = data.tier;                  // D  — Tier
  row[5]  = "New";                      // F  — Status
  row[12] = data.timestamp;             // M  — Date Added
  row[20] = "Consultation";             // U  — Workflow Type
  row[23] = data.phone;                 // X  — Phone
  row[24] = data.email;                 // Y  — Email
  row[26] = buildPipelineNotes(data);   // AA — Notes
  row[30] = "Website";                  // AE — Lead Source

  pipeline.appendRow(row);
  return id;
}

function reserveNextSkcId(ss, tier) {
  // Must use getScriptLock (not getDocumentLock) — standalone scripts
  // have no document binding, so getDocumentLock returns null.
  const lock = LockService.getScriptLock();
  lock.waitLock(10000);
  try {
    const settings = ss.getSheetByName("Settings");
    if (!settings) throw new Error("Settings sheet not found.");
    const rows = settings.getDataRange().getValues();
    let counterRow = -1;
    let currentNum = 500;

    rows.forEach((row, i) => {
      if (String(row[0]).includes("Next ID Counter") || row[0] === "ID Counter") {
        counterRow = i + 1;
        currentNum = parseInt(row[1], 10) || 500;
      }
    });

    const padded = String(currentNum).padStart(4, "0");
    const newId  = "SKC-" + tierCode(tier) + "-" + padded;

    if (counterRow > 0) {
      settings.getRange(counterRow, 2).setValue(currentNum + 1);
    } else {
      settings.appendRow(["ID Counter", currentNum + 1, "Auto-incremented — do not edit"]);
    }

    SpreadsheetApp.flush();
    return newId;
  } finally {
    lock.releaseLock();
  }
}

function buildPipelineNotes(data) {
  return "Source: " + data.formType + " form\n" +
         "Package selected: " + (data.serviceOrPackage || "(none)") + "\n" +
         "Vehicle of interest: " + (data.vehicle || "(none)") + "\n\n" +
         "Message:\n" + data.message;
}

// ── HELPERS ─────────────────────────────────────────────────
function normalizeFormData(params) {
  let name = String(params.name || "").trim();
  let formType = "consult";
  if (!name) {
    const first = String(params.first_name || "").trim();
    const last  = String(params.last_name  || "").trim();
    name = (first + " " + last).trim();
    formType = "contact";
  }
  return {
    name: name,
    email: String(params.email || "").trim(),
    phone: String(params.phone || "").trim(),
    serviceOrPackage: String(params.service || params.package || "").trim(),
    vehicle: String(params.vehicle || "").trim(),
    message: String(params.message || "").trim(),
    origin: String(params._origin || "").trim(),
    formType: formType,
  };
}

function validateLeadData(data) {
  const missing = [];
  if (!data.name) missing.push("name");
  if (!data.email) missing.push("email");
  if (!data.phone) missing.push("phone");
  if (!data.vehicle) missing.push("vehicle");
  if (!data.message) missing.push("message");
  if (missing.length) {
    return { ok: false, error: "Missing field: " + missing[0] };
  }
  if (!/^[^@\s]+@[^@\s]+\.[^@\s]+$/.test(data.email)) {
    return { ok: false, error: "Invalid email" };
  }
  return { ok: true };
}

function parseTier(serviceOrPackage) {
  const m = String(serviceOrPackage).match(/Tier\s*(\d)/i);
  if (m && (m[1] === "1" || m[1] === "2")) return "Tier " + m[1];
  return "Tier 1";
}

function tierCode(tier) {
  return tier === "Tier 2" ? "T2" : "T1";
}

function formatTimestamp(date) {
  return Utilities.formatDate(date, CONFIG.TIMEZONE, "yyyy-MM-dd HH:mm 'ET'");
}

function firstNameOf(fullName) {
  return String(fullName || "").trim().split(/\s+/)[0] || "Hi";
}

function escapeHtml(s) {
  return String(s || "")
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#39;");
}

function test_firstNameOf() {
  const cases = [
    ["Sarah Thompson", "Sarah"],
    ["Sarah", "Sarah"],
    ["  Sarah  Thompson  ", "Sarah"],
    ["Dr. Sarah Thompson", "Dr."],
    ["", "Hi"],
    [null, "Hi"],
  ];
  cases.forEach(pair => {
    const input = pair[0];
    const expected = pair[1];
    const got = firstNameOf(input);
    if (got !== expected) {
      throw new Error("firstNameOf(" + JSON.stringify(input) + ") => " + got + ", want " + expected);
    }
  });
  Logger.log("✅ test_firstNameOf passed (" + cases.length + " cases)");
}

function test_doPost_happyPath() {
  if (!CONFIG.TEST_CUSTOMER_EMAIL || CONFIG.TEST_CUSTOMER_EMAIL.indexOf("REPLACE") !== -1) {
    throw new Error("Set CONFIG.TEST_CUSTOMER_EMAIL first (top of file).");
  }
  const ss = SpreadsheetApp.openById(CONFIG.SHEET_ID);
  const pipeline = ss.getSheetByName("Pipeline");
  const beforeRows = pipeline.getLastRow();

  const fakeEvent = {
    parameter: {
      name: "_TEST_ Sarah Thompson",
      email: CONFIG.TEST_CUSTOMER_EMAIL,
      phone: "(555) 000-0000",
      service: "Tier 1 - Expert Advocate",
      vehicle: "Test Vehicle",
      message: "Happy-path integration test — safe to delete.",
      _origin: "https://skippr.us",
      company_website: "",
    }
  };
  const response = doPost(fakeEvent);
  const result = JSON.parse(response.getContent());
  if (!result.ok) throw new Error("Expected ok:true, got " + JSON.stringify(result));
  if (!result.id || result.id.indexOf("SKC-T1-") !== 0) {
    throw new Error("Expected id starting SKC-T1-, got " + result.id);
  }

  const afterRows = pipeline.getLastRow();
  if (afterRows !== beforeRows + 1) {
    throw new Error("Expected pipeline rows+1, before=" + beforeRows + " after=" + afterRows);
  }
  const newRow = pipeline.getRange(afterRows, 1, 1, 37).getValues()[0];
  if (newRow[0] !== result.id) throw new Error("Col A ID mismatch: " + newRow[0]);
  if (newRow[1] !== "_TEST_ Sarah Thompson") throw new Error("Col B name mismatch: " + newRow[1]);
  if (newRow[3] !== "Tier 1") throw new Error("Col D tier mismatch: " + newRow[3]);
  if (newRow[5] !== "New") throw new Error("Col F status mismatch: " + newRow[5]);
  if (newRow[20] !== "Consultation") throw new Error("Col U workflow mismatch: " + newRow[20]);
  if (newRow[24] !== CONFIG.TEST_CUSTOMER_EMAIL) throw new Error("Col Y email mismatch: " + newRow[24]);
  if (newRow[30] !== "Website") throw new Error("Col AE lead source mismatch: " + newRow[30]);

  pipeline.deleteRow(afterRows);
  Logger.log("✅ test_doPost_happyPath passed. ID: " + result.id + ". Two emails sent.");
}
